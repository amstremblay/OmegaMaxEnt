\section{Spectral representation for a bosonic Matsubara function} 

For a bosonic Matsubara function for which $\Lambda(\omega)=A(\omega)/\omega >0$,
\begin{equation}\label{eq:spectr_G}
\begin{split}
\chi(i\omega_n)&=\int_{-\infty}^{\infty}\frac{d\omega}{2\pi}\,\frac{A(\omega)}{i\omega_n-\omega}\\
&=\int_{-\infty}^{\infty}\frac{d\omega}{2\pi}\,\frac{\omega}{i\omega_n-\omega}\Lambda(\omega)\\
&=-\int_{-\infty}^{\infty}\frac{d\omega}{2\pi}\,\frac{\omega^2}{\omega_n^2+\omega^2}\Lambda(\omega)-i\omega_n\int_{-\infty}^{\infty}\frac{d\omega}{2\pi}\,\frac{\omega}{\omega_n^2+\omega^2}\Lambda(\omega)\\
\end{split}
\end{equation}

\begin{equation}
\text{Re}[\chi(i\omega_n)]=-\int_{-\infty}^{\infty}\frac{d\omega}{2\pi}\,\frac{\omega^2}{\omega_n^2+\omega^2}\Lambda(\omega)
\end{equation}
\begin{equation}
\text{Im}[\chi(i\omega_n)]=-\omega_n\int_{-\infty}^{\infty}\frac{d\omega}{2\pi}\,\frac{\omega}{\omega_n^2+\omega^2}\Lambda(\omega)
\end{equation}

If $\Lambda(\omega)=\Lambda(-\omega)$, then $\text{Im}[\chi(i\omega_n)]=0$.
